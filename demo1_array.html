# 배열과 반복문의 결합
# 배열에 원소 추가 1. push()
var li = ['a', 'b', 'c', 'd', 'e'];
li. push('f');
alert(li);
2. concat()
var li = ['a', 'b', 'c', 'd', 'e'];
li = li.concat(['f', 'g']);
alert(li);
3. unshift()
var li = ['a', 'b', 'c', 'd', 'e'];
li.unshift('z');
alert(li);
4. splice()
var li = ['a', 'b', 'c', 'd', 'e'];
li.splice(2, 0, 'B');
alert(li);
# 배열에서 원소 제거
1. shift
var li = ['a', 'b', 'c', 'd', 'e'];
li.shift();
alert(li);
2. pop
var li = ['a', 'b', 'c', 'd', 'e'];
li.pop();
alert(li);
# 정렬
1. sort()
var li = ['c', 'e', 'a', 'b', 'd'];
li.sort();
alert(li);
2. reverse()
var li = ['c', 'e', 'a', 'b', 'd'];
li.reverse();
alert(li);
